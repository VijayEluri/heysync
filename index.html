heysync by peter royal

transparent messaging passing in java using jetlang

Dependencies

Java 6, Jetlang, SLF4j

License

Apache License, Version 2.0

Authors

peter royal

Example

{% highlight java %} // Create our fiber ThreadFiber fiber = new ThreadFiber(); fiber.start(); // Channel for messages to flow on Channel messages = new MemoryChannel(); // Create our FieldMouse and subscribe him to the channel final FieldMouse fieldMouse = new FieldMouse(); messages.subscribe(fiber, new Callback() { @Override public void onMessage(String message) { fieldMouse.eatCheese(message); } }); // Tell our mouse to eat cheese messages.publish("cheddar"); fieldMouse.latch.await(); {% endhighlight %} {% highlight java %} // Create our fiber ThreadFiber fiber = new ThreadFiber(); fiber.start(); // Create heysync hub Hub hub = new Hub(); // Create our FieldMouse and add him as a receiver to our hub FieldMouse fieldMouse = new FieldMouse(); hub.addReceiver(fieldMouse, fiber); // Get the dispatcher for all registered Mice Mouse dispatcher = hub.dispatcherFor(Mouse.class); // Tell our mouse to eat cheese dispatcher.eatCheese("cheddar"); fieldMouse.latch.await(); {% endhighlight %}
